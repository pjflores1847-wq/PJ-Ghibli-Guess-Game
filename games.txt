  <script>
    const movies = [
      {title: "My Neighbor Totoro", img: "https://upload.wikimedia.org/wikipedia/en/0/02/My_Neighbor_Totoro_-_Tonari_no_Totoro_%28Movie_Poster%29.jpg"},
      {title: "Spirited Away", img: "https://upload.wikimedia.org/wikipedia/en/3/30/Spirited_Away_poster.JPG"},
      {title: "Howl's Moving Castle", img: "https://upload.wikimedia.org/wikipedia/en/a/a0/Howls-moving-castleposter.jpg"},
      {title: "Princess Mononoke", img: "https://upload.wikimedia.org/wikipedia/en/0/03/Princess_Mononoke_Japanese_Poster.jpg"},
      {title: "Kiki's Delivery Service", img: "https://upload.wikimedia.org/wikipedia/en/6/6f/Kiki%27s_Delivery_Service.jpg"}
    ];

    let currentIndex = 0;
    let options = [];

    function shuffle(array) {
      return array.sort(() => Math.random() - 0.5);
    }

    function showMovie() {
      const movie = movies[currentIndex];
      document.getElementById("movie-img").src = movie.img;
      document.getElementById("feedback").textContent = "";

      // Generate 3 options including correct answer
      const wrongChoices = movies.filter(m => m.title !== movie.title);
      options = shuffle([movie.title, wrongChoices[0].title, wrongChoices[1].title]);
      
      document.getElementById("choiceA").textContent = "A: " + options[0];
      document.getElementById("choiceB").textContent = "B: " + options[1];
      document.getElementById("choiceC").textContent = "C: " + options[2];
    }

    function checkGuess(choice) {
      const userGuess = options[{"A":0,"B":1,"C":2}[choice]];
      if (userGuess === movies[currentIndex].title) {
        document.getElementById("feedback").textContent = "‚úÖ Correct! Well done!";
      } else {
        document.getElementById("feedback").textContent = `‚ùå Incorrect! The correct title is: ${movies[currentIndex].title}`;
      }
    }

    function nextMovie() {
      currentIndex = (currentIndex + 1) % movies.length;
      showMovie();
    }

    // Initialize first movie
    showMovie();
  </script>
